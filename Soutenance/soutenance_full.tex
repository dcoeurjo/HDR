\begin{frame}
  \frametitle{Interval Arithmetic}


  \begin{block}{Elements = Intervals\HH}
    $x \sim  X= [\sI{x},\iI{x}]$ such that  $\sI{x}=  \fup{x}$  and
    $\iI{x}=\fdown{x}$  are \alert{representable numbers}, \emph{e.g.}
    \alert{$\I_{\Z+\frac{1}{2}}$}
  \end{block}



  \begin{block}{Examples and arithmetic operations on \alert{$\I_{\Z+\frac{1}{2}}$}\HH}
 \begin{align*}
   &3.144546\rightarrow [2+\frac{1}{2},3+\frac{1}{2}]\\
   &[2+\frac{1}{2},3+\frac{1}{2}] \alert{\oplus} [0+\frac{1}{2}, 4+\frac{1}{2}]
   = [2+\frac{1}{2}, 8+\frac{1}{2}]\\
&\ldots
 \end{align*}
  \end{block}

  \begin{alertblock}{Fundamental principle = Inclusion\HH}
    Let $f:\mathbb{R}\rightarrow\mathbb{R}$, and $\Box f:
    \mathbb{I}\rightarrow \mathbb{I}$ the interval function
    (\alert{extension}) associated to  $f$. For all $x\in X$, $\Box f$
    should be such that ~:
    \begin{displaymath}
      f(x)=y\quad \Rightarrow \quad y \in \Box f(X)
    \end{displaymath}
  \end{alertblock}
\end{frame}

\begin{block}{Voronoi Diagrams\HH}
  Given a set of sites ${\mathcal S}=\{s_i\}$ in ${\mathbb R}^2$,
  the Voronoi diagram is a decomposition of the plane into cells
  ${\mathcal C}=\{c_i\}$ (one cell $c_i \subset {\mathbb{R}}^2$ per
  site $s_i$) such that each point $p$ in the (open) cell $c_i$, we
  have $d(p,s_i)< d(p,s_j)$ for $i\neq j$.
\end{block}
  \column{3cm}
  \includegraphics[width=4cm]{voronoi_diagramme}

    \item DT algorithms $\rightarrow$ Discrete Voronoi Diagram mapping

      \item Association of each grid point to its associated cell
        $s_i$
      \item Difficulties to handle equidistant background points (when
        Voronoi diagram edges/vertices intersect $\Z^d$ points)
